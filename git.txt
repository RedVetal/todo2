# 1. Перейдите в папку вашего проекта (замените путь на актуальный)
cd путь/к/вашему/проекту

# 2. Инициализируйте Git в папке проекта
git init

# 3. Переименуйте ветку master в main (если Git создал master по умолчанию)
git branch -m master main

# 4. Добавьте все файлы проекта в staging area (область подготовки)
git add .

# 5. Создайте первый коммит с описанием изменений
git commit -m "Initial commit"

# 6. Свяжите локальный репозиторий с удаленным репозиторием на GitHub
# Замените ваш-username и название-репозитория на свои значения
git remote add origin https://github.com/ваш-username/название-репозитория.git

# 7. Отправьте локальную ветку main на GitHub
git push -u origin main

# 8. Дальнейшая работа с репозиторием:
# - Вносите изменения в файлы.
# - Добавляйте изменения в staging area:
#   git add .
# - Создавайте коммиты:
#   git commit -m "Описание изменений"
# - Отправляйте изменения на GitHub:
#   git push

# Полезные команды:
# - Проверка состояния репозитория:
#   git status
# - Просмотр истории коммитов:
#   git log
# - Проверка удаленных репозиториев:
#   git remote -v
